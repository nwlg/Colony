70      /** {@inheritDoc} */
71      public String getProfileId() {
72          return AttributeQueryConfiguration.PROFILE_ID;

75      /** {@inheritDoc} */
76      public void processRequest(HTTPInTransport inTransport, HTTPOutTransport outTransport) throws ProfileException {
77          AttributeQueryContext requestContext = new AttributeQueryContext();
78          Response samlResponse;
79          try {
80              decodeRequest(requestContext, inTransport, outTransport);
81  
82              if (requestContext.getProfileConfiguration() == null) {
83                  log.error("SAML 1 Attribute Query profile is not configured for relying party "
84                          + requestContext.getInboundMessageIssuer());
85                  requestContext.setFailureStatus(buildStatus(StatusCode.RESPONDER, StatusCode.REQUEST_DENIED,
86                          "SAML 1 Attribute Query profile is not configured for relying party "
87                                  + requestContext.getInboundMessageIssuer()));
88                  samlResponse = buildErrorResponse(requestContext);
89              } else {
90                  resolvePrincipal(requestContext);
91  
92                  Session idpSession = getSessionManager().getSession(requestContext.getPrincipalName());
93                  if (idpSession != null) {
94                      AuthenticationMethodInformation authnInfo = idpSession.getAuthenticationMethods().get(
95                              requestContext.getInboundMessageIssuer());
96                      if (authnInfo != null) {
97                          requestContext.setPrincipalAuthenticationMethod(authnInfo.getAuthenticationMethod());
98                      }
99                  }
100 
101                 resolveAttributes(requestContext);
102                 requestContext.setReleasedAttributes(requestContext.getAttributes().keySet());
103 
104                 ArrayList<Statement> statements = new ArrayList<Statement>();
105                 AttributeStatement attributeStatement = buildAttributeStatement(requestContext,
106                         "urn:oasis:names:tc:SAML:1.0:cm:sender-vouches");
107                 if (attributeStatement != null) {
108                     statements.add(attributeStatement);
109                 }
110 
111                 samlResponse = buildResponse(requestContext, statements);
112             }
113         } catch (ProfileException e) {
114             samlResponse = buildErrorResponse(requestContext);
115         }
116 
117         requestContext.setOutboundSAMLMessage(samlResponse);
118         requestContext.setOutboundSAMLMessageId(samlResponse.getID());
119         requestContext.setOutboundSAMLMessageIssueInstant(samlResponse.getIssueInstant());
120         encodeResponse(requestContext);
121         writeAuditLogEntry(requestContext);
122     }

124     /**
125      * Decodes an incoming request and populates a created request context with the resultant information.
126      * 
127      * @param inTransport inbound message transport
128      * @param outTransport outbound message transport
129      * @param requestContext the request context to which decoded information should be added
130      * 
131      * @throws ProfileException throw if there is a problem decoding the request
132      */
133     protected void decodeRequest(AttributeQueryContext requestContext, HTTPInTransport inTransport,
134             HTTPOutTransport outTransport) throws ProfileException {
135         if (log.isDebugEnabled()) {
136             log.debug("Decoding message with decoder binding {}",
137                     getInboundMessageDecoder(requestContext).getBindingURI());
138         }
139 
140         requestContext.setCommunicationProfileId(getProfileId());
141 
142         MetadataProvider metadataProvider = getMetadataProvider();
143         requestContext.setMetadataProvider(metadataProvider);
144 
145         requestContext.setInboundMessageTransport(inTransport);
146         requestContext.setInboundSAMLProtocol(SAMLConstants.SAML11P_NS);
147         requestContext.setSecurityPolicyResolver(getSecurityPolicyResolver());
148         requestContext.setPeerEntityRole(SPSSODescriptor.DEFAULT_ELEMENT_NAME);
149 
150         requestContext.setOutboundMessageTransport(outTransport);
151         requestContext.setOutboundSAMLProtocol(SAMLConstants.SAML11P_NS);
152 
153         try {
154             SAMLMessageDecoder decoder = getInboundMessageDecoder(requestContext);
155             requestContext.setMessageDecoder(decoder);
156             decoder.decode(requestContext);
157             log.debug("Decoded request");
158 
159             Request request = requestContext.getInboundSAMLMessage();
160             if (request == null || !(request instanceof Request) || request.getAttributeQuery() == null) {
161                 log.error("Incoming message was not an Attribute request");
162                 requestContext.setFailureStatus(buildStatus(StatusCode.REQUESTER, null,
163                         "Invalid SAML Attribute Request message."));
164                 throw new ProfileException("Invalid SAML Attribute Request message.");
165             }
166         } catch (MessageDecodingException e) {
167             log.warn("Error decoding attribute query message", e);
168             requestContext.setFailureStatus(buildStatus(StatusCode.RESPONDER, null, "Error decoding message"));
169             throw new ProfileException("Error decoding attribute query message", e);
170         } catch (SecurityException e) {
171             log.warn("Message did not meet security requirements", e);
172             requestContext.setFailureStatus(buildStatus(StatusCode.RESPONDER, StatusCode.REQUEST_DENIED,
173                     "Message did not meet security requirements"));
174             throw new ProfileException("Message did not meet security policy requirements", e);
175         } finally {
176             // Set as much information as can be retrieved from the decoded message
177             populateRequestContext(requestContext);
178         }
179     }

181     /** {@inheritDoc} */
182     protected void populateRelyingPartyInformation(BaseSAMLProfileRequestContext requestContext)
183             throws ProfileException {
184         super.populateRelyingPartyInformation(requestContext);
185 
186         EntityDescriptor relyingPartyMetadata = requestContext.getPeerEntityMetadata();
187         if (relyingPartyMetadata != null) {
188             requestContext.setPeerEntityRole(SPSSODescriptor.DEFAULT_ELEMENT_NAME);
189             requestContext.setPeerEntityRoleMetadata(relyingPartyMetadata.getSPSSODescriptor(SAMLConstants.SAML11P_NS));
190         }
191     }

193     /** {@inheritDoc} */
194     protected void populateAssertingPartyInformation(BaseSAMLProfileRequestContext requestContext)
195             throws ProfileException {
196         super.populateAssertingPartyInformation(requestContext);
197 
198         EntityDescriptor localEntityDescriptor = requestContext.getLocalEntityMetadata();
199         if (localEntityDescriptor != null) {
200             requestContext.setLocalEntityRole(AttributeAuthorityDescriptor.DEFAULT_ELEMENT_NAME);
201             requestContext.setLocalEntityRoleMetadata(localEntityDescriptor
202                     .getAttributeAuthorityDescriptor(SAMLConstants.SAML11P_NS));
203         }

206     /**
207      * Populates the request context with information from the inbound SAML message.
208      * 
209      * This method requires the the following request context properties to be populated: inbound saml message
210      * 
211      * This methods populates the following request context properties: subject name identifier
212      * 
213      * @param requestContext current request context
214      * 
215      * @throws ProfileException thrown if the inbound SAML message or subject identifier is null
216      */
217     protected void populateSAMLMessageInformation(BaseSAMLProfileRequestContext requestContext) throws ProfileException {
218         Request request = (Request) requestContext.getInboundSAMLMessage();
219         if (request == null) {
220             log.error("Decoder did not contain an attribute query, an error occured decoding the message");
221             throw new ProfileException("Unable to decode message.");
222         }
223 
224         AttributeQuery query = request.getAttributeQuery();
225         if (query != null) {
226             Subject subject = query.getSubject();
227             if (subject == null) {
228                 log.error("Attribute query did not contain a proper subject");
229                 ((AttributeQueryContext) requestContext).setFailureStatus(buildStatus(StatusCode.REQUESTER, null,
230                         "Attribute query did not contain a proper subject"));
231                 throw new ProfileException("Attribute query did not contain a proper subject");
232             }
233             requestContext.setSubjectNameIdentifier(subject.getNameIdentifier());
234         }
235     }

237     /**
238      * Selects the appropriate endpoint for the relying party and stores it in the request context.
239      * 
240      * @param requestContext current request context
241      * 
242      * @return Endpoint selected from the information provided in the request context
243      */
244     protected Endpoint selectEndpoint(BaseSAMLProfileRequestContext requestContext) {
245         Endpoint endpoint;
246 
247         if (getInboundBinding().equals(SAMLConstants.SAML1_SOAP11_BINDING_URI)) {
248             endpoint = acsEndpointBuilder.buildObject();
249             endpoint.setBinding(SAMLConstants.SAML1_SOAP11_BINDING_URI);
250         } else {
251             BasicEndpointSelector endpointSelector = new BasicEndpointSelector();
252             endpointSelector.setEndpointType(AssertionConsumerService.DEFAULT_ELEMENT_NAME);
253             endpointSelector.setMetadataProvider(getMetadataProvider());
254             endpointSelector.setEntityMetadata(requestContext.getPeerEntityMetadata());
255             endpointSelector.setEntityRoleMetadata(requestContext.getPeerEntityRoleMetadata());
256             endpointSelector.setSamlRequest(requestContext.getInboundSAMLMessage());
257             endpointSelector.getSupportedIssuerBindings().addAll(getSupportedOutboundBindings());
258             endpoint = endpointSelector.selectEndpoint();
259         }
260 
261         return endpoint;
262     }
